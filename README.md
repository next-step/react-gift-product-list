```
src/
├── components/
│   ├── atoms/           # 기본 UI 컴포넌트 (Button, Input, Text 등)
│   ├── molecules/       # 조합된 컴포넌트 (InputField, ItemCard 등)
│   └── organisms/       # 복잡한 UI 블록 (LoginForm, Navigation 등)
├── pages/              # 페이지 컴포넌트 (Home, Login, MyPage, Order)
├── contexts/           # 전역 상태 관리 (Auth, Receiver)
├── hooks/              # 커스텀 훅 (useInput, useOrderForm 등)
├── data/               # 정적 데이터 (orders, ranking, categories)
├── utils/              # 유틸리티 함수 (validation)
├── routes/             # 라우팅 설정 (router,protected)
└── styles/             # 스타일 정의 (spacing,typography,colors)
```
